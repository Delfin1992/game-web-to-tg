# Функции проверки модулей
def check_module(module_name):
    """Проверяет, установлен ли указанный модуль."""
    try:
        __import__(module_name)
        print(f"Модуль {module_name} найден.")
        return True
    except ImportError:
        print(f"Модуль {module_name} НЕ найден! Пожалуйста, установите его с помощью pip install {module_name}.")
        return False


# Проверка всех необходимых модулей
def check_required_modules():
    """Проверяет наличие всех необходимых модулей."""
    modules_to_check = ["sqlite3", "asyncio", "nest_asyncio", "telegram"]
    all_modules_found = True
    for module in modules_to_check:
        if not check_module(module):
            all_modules_found = False
    if not all_modules_found:
        print("Не все необходимые модули установлены. Установите их и перезапустите программу.")
        exit(1)  # Выход из программы, если хотя бы один модуль отсутствует
    print("Все необходимые модули установлены. Продолжаем запуск бота.")


# Подключение к базе данных SQLite
def init_db():
    """Инициализация базы данных."""
    conn = sqlite3.connect('players.db')
    cursor = conn.cursor()
    cursor.execute("""
        CREATE TABLE IF NOT EXISTS players (
            id INTEGER PRIMARY KEY,
            user_id INTEGER UNIQUE,
            name TEXT,
            city TEXT,
            personality TEXT,
            gram INTEGER DEFAULT 0,
            dollars INTEGER DEFAULT 0,  -- Валюта для Сан-Франциско
            won INTEGER DEFAULT 0,     -- Валюта для Сингапура и Сеула
            rubles INTEGER DEFAULT 0,  -- Валюта для Санкт-Петербурга
            coding INTEGER DEFAULT 5,
            testing INTEGER DEFAULT 5,
            analytics INTEGER DEFAULT 5,
            drawing INTEGER DEFAULT 5,
            modeling INTEGER DEFAULT 5,
            design INTEGER DEFAULT 5,
            attention INTEGER DEFAULT 5,
            level INTEGER DEFAULT 1,
            experience INTEGER DEFAULT 0,
            work_time REAL DEFAULT 8,  -- Время на работу (8 часов)
            study_time REAL DEFAULT 4  -- Время на учебу (4 часа)
        )
    """)
    # Проверяем наличие всех необходимых столбцов
    cursor.execute("PRAGMA table_info(players)")
    columns = [column[1] for column in cursor.fetchall()]
    required_columns = [
        "name", "city", "personality", "gram", "dollars", "won", "rubles",
        "coding", "testing", "analytics", "drawing", "modeling", "design",
        "attention", "level", "experience", "work_time", "study_time"
    ]
    for col in required_columns:
        if col not in columns:
            print(f"Добавляем столбец '{col}' в таблицу players.")
            cursor.execute(f"ALTER TABLE players ADD COLUMN {col} DEFAULT 0")
    conn.commit()
    conn.close()


# Функция для форматирования времени
def format_time(time_in_hours):
    """Форматирует время в виде 'X ч. Y мин.' или просто 'Y мин.'"""
    hours = int(time_in_hours)
    minutes = int((time_in_hours - hours) * 60)
    if hours > 0:
        return f"{hours} ч. {minutes} мин."
    else:
        return f"{minutes} мин."

# Фоновая задача для восстановления времени
async def continuous_restore_time():
    """Фоновая задача для восстановления времени для всех игроков."""
    while True:
        try:
            conn = sqlite3.connect('players.db')
            cursor = conn.cursor()
            # Обновляем время работы и учебы, если оно ниже максимума.
            cursor.execute("""
                UPDATE players
                SET work_time = CASE
                                  WHEN work_time < ? THEN MIN(work_time + ?, ?)
                                  ELSE work_time
                                END,
                    study_time = CASE
                                   WHEN study_time < ? THEN MIN(study_time + ?, ?)
                                   ELSE study_time
                                 END
            """, (
                MAX_WORK_TIME, RESTORE_INCREMENT, MAX_WORK_TIME,
                MAX_STUDY_TIME, RESTORE_INCREMENT, MAX_STUDY_TIME,
            ))
            conn.commit()
        except Exception as e:
            print("Ошибка при восстановлении времени:", e)
        finally:
            conn.close()
        await asyncio.sleep(1)


# Обработчик команды /start
async def start(update: Update, context: CallbackContext) -> int:
    """Запускает бота. Если игрок уже существует, показывает его профиль."""
    user = update.effective_user
    conn = sqlite3.connect('players.db')
    cursor = conn.cursor()
    try:
        # Проверяем, есть ли игрок в базе
        cursor.execute("SELECT name, city, level FROM players WHERE user_id = ?", (user.id,))
        player = cursor.fetchone()
        if player:
            name, city, level = player
            # Исправление уровня, если он равен нулю
            if level < 1:
                level = 1
                cursor.execute("UPDATE players SET level = 1 WHERE user_id = ?", (user.id,))
                conn.commit()
            # Игрок уже существует — загружаем данные в context.user_data
            context.user_data["name"] = name
            context.user_data["city"] = city
            context.user_data["level"] = level
            await update.message.reply_text("С возвращением! Вот ваш профиль:")
            await show_profile(update, context)
            return SHOW_PROFILE
        else:
            # Игрока нет в базе — запускаем регистрацию
            await update.message.reply_text("Добро пожаловать! Давай познакомимся. Как тебя зовут?")
            return GET_NAME
    finally:
        conn.close()


async def save_name(update: Update, context: CallbackContext) -> int:
    """Сохраняет имя игрока и предлагает выбрать город."""
    user = update.effective_user
    name = update.message.text.strip()
    # Сохраняем имя в базу данных вместе с начальным временем
    conn = sqlite3.connect('players.db')
    cursor = conn.cursor()
    try:
        # Проверяем, существует ли игрок
        cursor.execute("SELECT user_id FROM players WHERE user_id = ?", (user.id,))
        player_exists = cursor.fetchone()
        if player_exists:
            # Если игрок существует, обновляем время, если оно не установлено
            cursor.execute("""
                UPDATE players
                SET work_time = 0.5, study_time = 0.5
                WHERE user_id = ? AND (work_time IS NULL OR study_time IS NULL )
            """, (user.id,))
        else:
            # Если игрок не существует, добавляем его с начальными значениями
            cursor.execute("""
                INSERT INTO players (
                    user_id, name, work_time, study_time, level
                ) VALUES (?, ?, ?, ?, ?)
            """, (user.id, name, 0.5, 0.5, 1))  # Устанавливаем level = 1
        conn.commit()
    finally:
        conn.close()
    # Отправляем информацию о переходе к выбору города
    await update.message.reply_text(f"Приятно познакомиться, {name}! Теперь выбери город, в котором ты хочешь жить.")
    # Отправляем клавиатуру для выбора города
    await show_city_options(update, context)
    return SELECT_CITY

async def work(update: Update, context: CallbackContext) -> int:
    """Обрабатывает команду 'Работа'."""
    user = update.effective_user
    conn = sqlite3.connect('players.db')
    cursor = conn.cursor()

    try:
        # Загружаем информацию о городе и уровне игрока
        cursor.execute("SELECT city, level FROM players WHERE user_id = ?", (user.id,))
        result = cursor.fetchone()

        if not result:
            await update.message.reply_text("Не удалось загрузить ваш профиль. Попробуйте снова.")
            return SHOW_PROFILE

        city, level = result
        context.user_data["city", ""] = city
        context.user_data["level",1] = level

        # Логирование загруженных данных
        print(f"City: {city}, Level: {level}")

        available_jobs = get_available_jobs(city, level)

        if not available_jobs:
            await update.message.reply_text(
                f"В данный момент нет доступных работ для вашего уровня ({level}) в городе {city}."
            )
            return SHOW_PROFILE
        # Формирование кнопок клавиатуры
        keyboard = [[job["name"]] for job in available_jobs] + [["Домой"]]
        reply_markup = ReplyKeyboardMarkup(keyboard, resize_keyboard=True)

        await update.message.reply_text(f"Выберите работу в городе {city}:", reply_markup=reply_markup)
        return WORK_SELECTION

    finally:
        conn.close()

async def handle_work_selection(update: Update, context: CallbackContext) -> int:
    """Обрабатывает выбор работы и обновляет профиль игрока."""
    selected_job = update.message.text.strip()
    city = context.user_data.get("city", "Неизвестный город")
    level = context.user_data.get("level", 1)
    if selected_job == "Домой":
        await show_profile(update, context)
        return SHOW_PROFILE

    available_jobs = get_available_jobs(city, level)
    job_info = next((job for job in available_jobs if job["name"] == selected_job), None)
    if not job_info:

# Обработка выбранного курса
async def handle_study_course(update: Update, context: CallbackContext) -> int:
    """Обрабатывает выбор учебного курса."""
    user = update.effective_user
    course_name = update.message.text.strip()

    if course_name == "Назад":
        return await study(update, context)

    selected_level = context.user_data.get("selected_study_level")

    if not selected_level or course_name not in STUDY_COSTS[selected_level]:
        await update.message.reply_text("Неверный выбор. Попробуйте снова.")
        return STUDY

    course = STUDY_COSTS[selected_level][course_name]

    conn = sqlite3.connect('players.db')
    cursor = conn.cursor()
    try:
        cursor.execute("SELECT study_time, dollars, coding, analytics FROM players WHERE user_id = ?", (user.id,))
        current_study_time, current_balance, coding, analytics = cursor.fetchone()

        if current_balance < course["cost"]:
            await update.message.reply_text(f"Недостаточно средств. Нужно {course['cost']} $. У вас {current_balance} $.")
            return STUDY

        if current_study_time < course["time_cost"]:
            await update.message.reply_text(f"Недостаточно времени для учебы. У вас есть {current_study_time} ч.")
            return STUDY

        # Проверка на провал курса
        if random.random() < course["fail_rate"]:
            await update.message.reply_text("Курс завершился неудачно. Попробуйте снова.")
            return STUDY

        # Применяем изменения навыков и обновляем данные
        new_study_time = current_study_time - course["time_cost"]
        new_balance = current_balance - course["cost"]

        coding_gain = course.get("skill_gain", {}).get("coding", 0)
        analytics_gain = course.get("skill_gain", {}).get("analytics", 0)

        cursor.execute(
            "UPDATE players SET study_time = ?, dollars = ?, coding = coding + ?, analytics = analytics + ? WHERE user_id = ?",
            (new_study_time, new_balance, coding_gain, analytics_gain, user.id)
        )
        conn.commit()

        # Проверяем уровень игрока
        await check_level_up(update, context)

        # Отправляем информацию о выполненном курсе
        await update.message.reply_text(
            f"Вы успешно завершили курс '{course_name}'.\n"
            f"Затрачено времени: {course['time_cost']} ч.\n"
            f"Затрачено средств: {course['cost']} $\n"
            f"Получено навыков: Кодинг +{coding_gain}, Аналитика +{analytics_gain}\n"
            f"Оставшееся время на учебу: {new_study_time} ч."
        )
    finally:
        conn.close()

    # Показываем профиль игрока
    await show_profile(update, context)
    return SHOW_PROFILE

# Получение бонусов за новый уровень
def get_level_up_bonuses(level: int, chosen_stat: str) -> dict:
    """Возвращает бонусы за новый уровень и увеличивает выбранную характеристику."""
    # Список характеристик, которые игрок может выбрать
    stats = ["coding", "testing", "analytics", "drawing", "modeling", "design", "attention"]

    # Начальные бонусы на определенные уровни
    bonuses = {
        25: {"coding": 5, "testing": 5},
        50: {"analytics": 5, "drawing": 5},
        100: {"modeling": 5, "design": 5},
        150: {"attention": 5},
        200: {"coding": 10, "testing": 10},
        250: {"analytics": 10, "drawing": 10},
        300: {"modeling": 10, "design": 10},
        350: {"attention": 10},
        400: {"coding": 15, "testing": 15},
        450: {"analytics": 15, "drawing": 15},
        500: {"modeling": 15, "design": 15, "attention": 15}
    }

    # Проверка, что выбранная характеристика корректна
    if chosen_stat not in stats:
        raise ValueError(f"Invalid stat chosen: {chosen_stat}. Choose from {', '.join(stats)}.")

    # Бонусы за уровень, начиная с уровня 1 (по 1 очку на характеристику)
    level_bonuses = {stat: 0 for stat in stats}

    # Увеличиваем выбранную характеристику на 1
    level_bonuses[chosen_stat] = 1

    # Обновление времени на работу и учебу
    work_study_time = 20  # начальное время
    if level >= 100:
        work_study_time = 50  # до 100 уровня
    elif level >= 50:
        work_study_time = 40  # до 50 уровня
    elif level >= 9:
        work_study_time = 30  # до 9 уровня

    # Объединение бонусов на конкретный уровень с выбранной характеристикой
    combined_bonuses = {**bonuses.get(level, {}), **level_bonuses}

    return combined_bonuses, work_study_time

# Удаление игрока
async def delete_me(update: Update, context: CallbackContext):
    """Полностью удаляет игрока из базы данных и сбрасывает его данные в памяти бота."""
    user = update.effective_user
    conn = sqlite3.connect('players.db')
    cursor = conn.cursor()
    try:
        # Проверяем, существует ли игрок
        cursor.execute("SELECT user_id FROM players WHERE user_id = ?", (user.id,))
        player = cursor.fetchone()
        if player:
            # Удаляем игрока из базы
            cursor.execute("DELETE FROM players WHERE user_id = ?", (user.id,))
            conn.commit()
            # Очищаем данные игрока в `context.user_data`
            context.user_data.clear()
            await update.message.reply_text("Ваш аккаунт был полностью удалён. Чтобы начать заново, введите /start.")
        else:
            await update.message.reply_text("Ваш аккаунт не найден в базе данных.")
    finally:
        conn.close()

    # Возвращаем END для полного завершения диалога
    return ConversationHandler.END

# Обновление характеристик игрока
def update_player_characteristics(user_id: int, personality: str) -> None:
    """Обновляет характеристики игрока в зависимости от выбранного характера."""
    conn = sqlite3.connect('players.db')
    cursor = conn.cursor()
    try:
        # Извлекаем текущие характеристики игрока
        cursor.execute("""
            SELECT coding, testing, analytics, drawing, modeling, design, attention 
            FROM players WHERE user_id = ?
        """, (user_id,))
        player = cursor.fetchone()

        if player:
            # Преобразуем все характеристики к целочисленному типу
            current_characteristics = {
                "coding": int(player[0]),
                "testing": int(player[1]),
                "analytics": int(player[2]),
                "drawing": int(player[3]),
                "modeling": int(player[4]),
                "design": int(player[5]),
                "attention": int(player[6])
            }

            effects = PERSONALITIES.get(personality, {})
            for characteristic, value in effects.items():
                if characteristic in current_characteristics:
                    current_value = current_characteristics[characteristic]
                    new_value = current_value + value  # Здесь больше не будет TypeError
                    cursor.execute(f"UPDATE players SET {characteristic} = ? WHERE user_id = ?", (new_value, user_id))

            # Обновляем мультипликаторы опыта и денег
            if "experience_multiplier" in effects:
                cursor.execute("UPDATE players SET experience_multiplier = ? WHERE user_id = ?",
                               (effects["experience_multiplier"], user_id))
            if "money_multiplier" in effects:
                cursor.execute("UPDATE players SET money_multiplier = ? WHERE user_id = ?",
                               (effects["money_multiplier"], user_id))

        conn.commit()
    finally:
        conn.close()

async def handle_stat_selection(update: Update, context: CallbackContext) -> None:
    """Добавляет выбранную характеристику игроку и убирает клавиатуру.
    """
    query = update.callback_query
    await query.answer()  # Подтверждаем нажатие кнопки
    chosen_stat = query.data  # Получаем выбранную характеристику
    user_id = query.from_user.id  # ID игрока

    # Обновляем БД, увеличивая выбранную характеристику на 1
    conn = sqlite3.connect('players.db')
    cursor = conn.cursor()
    try:
        cursor.execute("SELECT * FROM players WHERE user_id = ?", (user_id,))
        player = cursor.fetchone()
        if player:
            cursor.execute(f"UPDATE players SET {chosen_stat} = {chosen_stat} + 1 WHERE user_id = ?", (user_id,))
            conn.commit()
    finally:
        conn.close()

    translated_stat = STAT_TRANSLATIONS.get(chosen_stat, chosen_stat)

    # Убираем клавиатуру, заменяя сообщение
    await query.edit_message_text(f"🎉 Ваша характеристика {translated_stat} увеличена на 1!")
    # Показываем профиль игрока после выбора характеристики
    await show_profile(update, context)
